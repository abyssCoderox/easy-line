## 1. 技术选型说明

### 1.1 核心技术栈

| 层级 | 技术选型 | 版本 | 选型理由 |
|------|---------|------|---------|
| 运行时 | Node.js | 20+ | LTS版本，官方SDK要求 |
| 开发语言 | TypeScript | 5.x | 类型安全，开发体验好 |
| Web框架 | Express | 4.x | 轻量级，官方SDK示例一致 |
| LINE SDK | @line/bot-sdk | 7.x | LINE官方SDK，功能完整 |
| 定时任务 | node-cron | 3.x | 轻量级Cron调度器 |
| LLM框架 | LangChain.js | 0.1.x | 成熟框架，快速集成大模型 |
| 环境变量 | dotenv | 16.x | 配置管理 |
| HTTP客户端 | axios | 1.x | 第三方API调用 |

### 1.2 技术选型对比

#### 1.2.1 Web框架选择

| 框架 | 优势 | 劣势 | 结论 |
|------|------|------|------|
| **Express** ✅ | 轻量、官方示例一致、学习成本低 | 无内置高级功能 | **选择** |
| NestJS | 模块化、企业级 | 学习成本高、Demo过度设计 | 不适合Demo |
| Fastify | 高性能 | 生态较小 | 不必要 |
| Koa | async/await优雅 | 需要更多中间件 | 可选 |

#### 1.2.2 定时任务选择

| 方案 | 优势 | 劣势 | 结论 |
|------|------|------|------|
| **node-cron** ✅ | 轻量、简单、Cron语法 | 单进程 | **选择** |
| BullMQ | 功能强大、持久化 | 需要Redis | Demo过度 |
| Agenda | MongoDB集成 | 需要数据库 | 不需要 |
| node-schedule | 灵活调度 | API稍复杂 | 可选 |

#### 1.2.3 为什么选择@line/bot-sdk

| 理由 | 说明 |
|------|------|
| ✅ 官方维护 | LINE官方维护，API更新及时 |
| ✅ 功能完整 | 覆盖所有Messaging API功能 |
| ✅ TypeScript原生 | 完整类型定义 |
| ✅ 文档完善 | 官方文档和示例丰富 |
| ✅ 中间件支持 | 内置签名验证中间件 |

## 4. 核心模块设计

### 4.1 LINE服务模块 (LineService)

#### 职责
- 封装@line/bot-sdk客户端
- 提供消息接收、回复、推送功能
- 处理签名验证

#### 类设计

```typescript
// src/services/line.service.ts
import { messagingApi, middleware, MiddlewareConfig, webhook } from '@line/bot-sdk';

export class LineService {
  private client: messagingApi.MessagingApiClient;
  private middlewareConfig: MiddlewareConfig;
  
  constructor(config: LineConfig) {
    this.client = new messagingApi.MessagingApiClient({
      channelAccessToken: config.channelAccessToken,
    });
    this.middlewareConfig = {
      channelSecret: config.channelSecret,
    };
  }
  
  // 获取签名验证中间件
  getMiddleware() {
    return middleware(this.middlewareConfig);
  }
  
  // 回复消息
  async replyMessage(replyToken: string, messages: Message[]) {
    return this.client.replyMessage({
      replyToken,
      messages,
    });
  }
  
  // 推送消息 ⭐
  async pushMessage(to: string, messages: Message[]) {
    return this.client.pushMessage({
      to,
      messages,
    });
  }
  
  // 批量推送消息 ⭐
  async multicastMessage(to: string[], messages: Message[]) {
    return this.client.multicast({
      to,
      messages,
    });
  }
  
  // 获取用户信息
  async getUserProfile(userId: string) {
    return this.client.getProfile(userId);
  }
}

export const lineService = new LineService({
  channelAccessToken: process.env.CHANNEL_ACCESS_TOKEN!,
  channelSecret: process.env.CHANNEL_SECRET!,
});
```

### 4.2 定时任务服务模块 (SchedulerService) ⭐

#### 职责
- 管理定时任务配置
- 执行Cron调度
- 调用第三方API获取数据
- 推送消息到LINE

#### 类设计

```typescript
// src/services/scheduler.service.ts
import cron from 'node-cron';
import { lineService } from './line.service';
import { apiService } from './api.service';

interface TaskConfig {
  id: string;
  name: string;
  cronExpression: string;
  apiEndpoint: string;
  apiMethod: 'GET' | 'POST';
  apiHeaders?: Record<string, string>;
  apiBody?: any;
  messageTemplate: string;
  targetType: 'user' | 'group';
  targetIds: string[];
  enabled: boolean;
}

interface TaskLog {
  taskId: string;
  executeTime: Date;
  status: 'success' | 'failed';
  apiResponse?: any;
  pushResult?: any;
  error?: string;
}

export class SchedulerService {
  private tasks: Map<string, cron.ScheduledTask> = new Map();
  private taskConfigs: TaskConfig[] = [];
  private logs: TaskLog[] = [];
  
  constructor() {
    this.loadTaskConfigs();
  }
  
  // 加载任务配置
  private loadTaskConfigs() {
    // 从配置文件或环境变量加载
    this.taskConfigs = require('../config/tasks.json');
  }
  
  // 启动所有任务
  startAll() {
    this.taskConfigs
      .filter(config => config.enabled)
      .forEach(config => this.startTask(config));
    
    console.log(`Started ${this.tasks.size} scheduled tasks`);
  }
  
  // 启动单个任务
  startTask(config: TaskConfig) {
    if (!cron.validate(config.cronExpression)) {
      console.error(`Invalid cron expression: ${config.cronExpression}`);
      return;
    }
    
    const task = cron.schedule(config.cronExpression, async () => {
      await this.executeTask(config);
    });
    
    this.tasks.set(config.id, task);
    console.log(`Task started: ${config.name} (${config.cronExpression})`);
  }
  
  // 执行任务 ⭐
  private async executeTask(config: TaskConfig): Promise<void> {
    const startTime = Date.now();
    const log: TaskLog = {
      taskId: config.id,
      executeTime: new Date(),
      status: 'success',
    };
    
    try {
      // 1. 调用第三方API获取数据
      const apiResponse = await apiService.call({
        url: config.apiEndpoint,
        method: config.apiMethod,
        headers: config.apiHeaders,
        body: config.apiBody,
      });
      log.apiResponse = apiResponse;
      
      // 2. 数据转换处理
      const message = this.transformData(apiResponse, config.messageTemplate);
      
      // 3. 推送消息
      if (config.targetType === 'user') {
        for (const userId of config.targetIds) {
          const result = await lineService.pushMessage(userId, [message]);
          log.pushResult = result;
        }
      } else {
        const result = await lineService.multicastMessage(config.targetIds, [message]);
        log.pushResult = result;
      }
      
      console.log(`Task executed successfully: ${config.name}`);
      
    } catch (error: any) {
      log.status = 'failed';
      log.error = error.message;
      console.error(`Task execution failed: ${config.name}`, error);
    }
    
    // 4. 记录日志
    this.logs.push(log);
  }
  
  // 数据转换
  private transformData(data: any, template: string): any {
    // 根据模板转换数据为LINE消息格式
    const text = template.replace(/\{(\w+)\}/g, (_, key) => data[key] || '');
    return { type: 'text', text };
  }
  
  // 停止任务
  stopTask(taskId: string) {
    const task = this.tasks.get(taskId);
    if (task) {
      task.stop();
      this.tasks.delete(taskId);
    }
  }
  
  // 停止所有任务
  stopAll() {
    this.tasks.forEach(task => task.stop());
    this.tasks.clear();
  }
  
  // 获取任务状态
  getTaskStatus(): { id: string; running: boolean }[] {
    return this.taskConfigs.map(config => ({
      id: config.id,
      running: this.tasks.has(config.id),
    }));
  }
  
  // 获取执行日志
  getLogs(limit: number = 100): TaskLog[] {

}
```

### 6.3 Webhook路由实现

```typescript
// src/routes/webhook.ts
import { Router } from 'express';
import { lineService } from '../services/line.service';
import { llmService } from '../services/llm.service';
import { webhook } from '@line/bot-sdk';

const router = Router();

// 应用LINE签名验证中间件
router.post('/', lineService.getMiddleware(), async (req, res) => {
  try {
    const events = req.body.events;
    
    // 并行处理所有事件
    await Promise.all(events.map(handleEvent));
    
    res.json({ status: 'ok' });
  } catch (error) {
    console.error('Webhook error:', error);
    res.status(500).json({ status: 'error' });
  }
});

// 事件处理函数
async function handleEvent(event: webhook.Event) {
  if (event.type !== 'message') return;
  if (event.message.type !== 'text') return;
  
  const userId = event.source.userId!;
  const userMessage = event.message.text;
  const replyToken = event.replyToken;
  
  // 调用LLM生成回复
  const reply = await llmService.chat(userId, userMessage);
  
  // 发送回复
  await lineService.replyMessage(replyToken, [{
    type: 'text',
    text: reply,
  }]);
}

export default router;
```
